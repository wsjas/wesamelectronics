rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null && request.auth.token.admin == true;
    }

    // Public read
    match /categories/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /products/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // Orders: anyone can create, only admin can read/update
    match /orders/{id} {
      allow create: if true;
      allow read, update, delete: if isAdmin();
    }
  }
}

Notes:
- To use token.admin, set a custom claim on your admin user via Firebase Admin SDK.
- If you don't want custom claims, you can switch to a users collection + allow by uid.
